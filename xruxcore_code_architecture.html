SessionManager

Singleton. Generates session_id, tracks phase, carries profile across scenes.
SingletonDontDestroyOnLoad
public class SessionManager : MonoBehaviour
{
    public static SessionManager Instance;

    // ── Fields ──────────────────────────────────────
    [HideInInspector] public string      sessionId;
    [HideInInspector] public int         participantId;
    [HideInInspector] public string      currentPhase;  // tutorial|sorting|procedural
    [HideInInspector] public float       sessionStartTime;
    [HideInInspector] public float       phaseStartTime;

    public UXProfileSO activeProfile; // set before Play in Inspector

    // ── Unity lifecycle ──────────────────────────────
    void Awake()
    {
        if (Instance != null) { Destroy(gameObject); return; }
        Instance = this;
        DontDestroyOnLoad(gameObject);
        sessionId = System.Guid.NewGuid().ToString();
        sessionStartTime = Time.time;
    }

    void Start() => LogSessionStart();

    // ── Public API ───────────────────────────────────
    public void TransitionToPhase(string phase)
    {
        currentPhase = phase;
        phaseStartTime = Time.time;
        TelemetryManager.Instance.LogEvent("phase_transition",
            new { phase, timestamp = Time.time });
    }

    public float GetPhaseDuration() => Time.time - phaseStartTime;
    public float GetSessionDuration() => Time.time - sessionStartTime;

    private void LogSessionStart()
    {
        TelemetryManager.Instance.LogEvent("session_start", new {
            participantId,
            profile = activeProfile.profileName,
            isOptimized = activeProfile.isOptimized
        });
    }
}

Core / TelemetryManager.cs

TelemetryManager

Singleton. Routes events to MongoDB via REST. Logs HighFreq at 60 Hz.
SingletonCoroutine
public class TelemetryManager : MonoBehaviour
{
    public static TelemetryManager Instance;

    // ── Config ──────────────────────────────────────
    [SerializeField] string apiBaseUrl;   // your MongoDB Data API endpoint
    [SerializeField] string apiKey;        // stored in SO or env
    [SerializeField] float  highFreqHz = 60f;

    // ── Private state ────────────────────────────────
    private float _nextHighFreqTime;
    private Queue<string> _eventQueue = new();

    void Update()
    {
        if (Time.time >= _nextHighFreqTime)
        {
            _nextHighFreqTime = Time.time + (1f / highFreqHz);
            CaptureHighFreqFrame();
        }
    }

    // ── Discrete events ──────────────────────────────
    public void LogEvent(string eventType, object data)
    {
        var payload = new {
            session_id  = SessionManager.Instance.sessionId,
            timestamp   = Time.time,
            phase       = SessionManager.Instance.currentPhase,
            event_type  = eventType,
            data
        };
        StartCoroutine(PostToMongo("Events", payload));
    }

    // ── High-frequency frame ─────────────────────────
    private void CaptureHighFreqFrame()
    {
        var cam = Camera.main.transform;
        var payload = new {
            session_id  = SessionManager.Instance.sessionId,
            timestamp   = Time.time,
            event_type  = "head_tracking",
            data = new {
                position = cam.position,
                rotation = cam.rotation.eulerAngles,
                pitch    = cam.rotation.eulerAngles.x,
                yaw      = cam.rotation.eulerAngles.y
            }
        };
        StartCoroutine(PostToMongo("HighFreq_Logs", payload));
    }

    // ── Snap precision events ────────────────────────
    public void LogSnapEvent(bool success, float distErr,
                             float angErr, string objectId)
    {
        LogEvent(success ? "snap_success" : "snap_fail", new {
            objectId, distanceError = distErr, angularError = angErr
        });
    }

    // ── HTTP helper ──────────────────────────────────
    private IEnumerator PostToMongo(string collection, object body)
    {
        string json = JsonUtility.ToJson(body);
        using var req = new UnityWebRequest(apiBaseUrl + collection, "POST");
        req.uploadHandler   = new UploadHandlerRaw(System.Text.Encoding.UTF8.GetBytes(json));
        req.downloadHandler = new DownloadHandlerBuffer();
        req.SetRequestHeader("Content-Type", "application/json");
        req.SetRequestHeader("api-key", apiKey);
        yield return req.SendWebRequest();
    }
}

XRRigConfigurator

Reads UXProfileSO on Start and toggles all XR components accordingly.
public class XRRigConfigurator : MonoBehaviour
{
    // ── Inspector refs ──────────────────────────────
    [Header("Profile")]
    public UXProfileSO profile; // drag Baseline or Optimized asset here

    [Header("Locomotion")]
    public GameObject teleportProvider;
    public GameObject continuousMoveProvider;

    [Header("Input")]
    public GameObject handTrackingLeft;
    public GameObject handTrackingRight;
    public GameObject controllerLeft;
    public GameObject controllerRight;

    [Header("UI")]
    public GameObject fixedHUD;
    public GameObject wristUILeft;

    [Header("Comfort")]
    public TunnelingVignetteController vignetteController;

    void Start()
    {
        ApplyProfile();
        TelemetryManager.Instance.LogEvent("ux_profile_applied",
            new { profile = profile.profileName });
    }

    private void ApplyProfile()
    {
        // Locomotion
        teleportProvider.SetActive(profile.teleportOnly || profile.hybridLocomotion);
        continuousMoveProvider.SetActive(profile.hybridLocomotion);

        // Input mode
        handTrackingLeft.SetActive(profile.useHandTracking);
        handTrackingRight.SetActive(profile.useHandTracking);
        controllerLeft.SetActive(profile.useController);
        controllerRight.SetActive(profile.useController);

        // UI mode
        fixedHUD.SetActive(profile.useFixedHUD);
        wristUILeft.SetActive(profile.useDiegeticWristUI);

        // Comfort vignette
        if (vignetteController != null)
            vignetteController.enabled = profile.enableComfortVignette;

        // Log mode_change event
        TelemetryManager.Instance.LogEvent("ui_mode_change", new {
            uiMode = profile.useDiegeticWristUI ? "diegetic" : "fixed_hud",
            locomotion = profile.teleportOnly ? "teleport" : "hybrid",
            input = profile.useHandTracking ? "hand" : "controller"
        });
    }
}

SnapInteractor

Attach to every grabbable object. Calculates distance/angular error on release, snaps if profile allows, logs all precision metrics.
XRGrabInteractableLogs snap_success / snap_fail
public class SnapInteractor : MonoBehaviour
{
    // ── Inspector ────────────────────────────────────
    public Transform  snapTarget;       // empty GameObject at ideal placement
    public string     objectId;         // e.g. "canopic_jar_01"
    public AudioClip  snapSFX;
    public GameObject glowEffect;

    // ── Runtime state ────────────────────────────────
    private float    _grabTimestamp;
    private int      _grabAttempts;
    private float    _distanceError;
    private float    _angularError;
    private bool     _isSnapped;

    // ── Profile reference (from SessionManager) ──────
    private UXProfileSO _profile =>
        SessionManager.Instance.activeProfile;

    // ── XRI callbacks ───────────────────────────────
    public void OnGrabbed()
    {
        _grabTimestamp = Time.time;
        _grabAttempts++;
        TelemetryManager.Instance.LogEvent("grab", new { objectId });
    }

    public void OnReleased()
    {
        float holdDuration = Time.time - _grabTimestamp;
        CalculatePlacementErrors();

        bool snapSuccess = _profile.snappingEnabled
            && _distanceError <= _profile.snapDistanceThreshold
            && _angularError  <= _profile.snapAngleThreshold;

        if (snapSuccess) ExecuteSnap();

        TelemetryManager.Instance.LogSnapEvent(
            snapSuccess, _distanceError, _angularError, objectId);

        TelemetryManager.Instance.LogEvent("release", new {
            objectId, holdDuration, snapSuccess,
            distanceError = _distanceError,
            angularError  = _angularError,
            attemptNumber = _grabAttempts
        });

        TriggerFeedback(snapSuccess);
    }

    // ── Helpers ──────────────────────────────────────
    private void CalculatePlacementErrors()
    {
        if (snapTarget == null) return;
        _distanceError = Vector3.Distance(transform.position, snapTarget.position);
        _angularError  = Quaternion.Angle(transform.rotation, snapTarget.rotation);
    }

    private void ExecuteSnap()
    {
        transform.position = snapTarget.position;
        transform.rotation = snapTarget.rotation;
        _isSnapped = true;
    }

    private void TriggerFeedback(bool success)
    {
        FeedbackManager.Instance.Trigger(
            haptic:   _profile.useHaptics,
            audio:    _profile.useAudioFeedback ? snapSFX : null,
            glow:     _profile.useVisualGlow ? glowEffect : null,
            intensity:_profile.feedbackIntensity,
            success:  success
        );
    }
}
Core / FeedbackManager.cs

FeedbackManager

Central hub for haptics, audio, and visual glow. Called by SnapInteractor and room managers.
public class FeedbackManager : MonoBehaviour
{
    public static FeedbackManager Instance;
    public AudioSource audioSource;
    public XRBaseController leftController, rightController;

    public void Trigger(bool haptic, AudioClip audio,
                        GameObject glow, float intensity, bool success)
    {
        if (haptic)
        {
            float amp = success ? intensity : intensity * 0.4f;
            leftController.SendHapticImpulse(amp, 0.1f);
            rightController.SendHapticImpulse(amp, 0.1f);
        }
        if (audio != null)
            audioSource.PlayOneShot(audio, intensity);

        if (glow != null)
            StartCoroutine(FlashGlow(glow, intensity));
    }

    private IEnumerator FlashGlow(GameObject g, float intensity)
    {
        g.SetActive(true);
        yield return new WaitForSeconds(0.3f * intensity);
        g.SetActive(false);
    }
}
Room 1 Tutorial / TutorialManager.cs

TutorialManager

Orchestrates 3 tutorial steps: movement → shovel grab → door open. Logs each completion.
public class TutorialManager : MonoBehaviour
{
    // ── Steps ────────────────────────────────────────
    public enum TutStep { MoveTo, GrabShovel, OpenDoor, Complete }
    public TutStep currentStep = TutStep.MoveTo;

    // ── Inspector refs ──────────────────────────────
    public Transform      movementTarget; // trigger zone at tomb entrance
    public ShovelInteractable shovel;
    public TombDoorOpener    door;
    public TaskInstructionUI instructionUI;

    // ── Metrics ──────────────────────────────────────
    private float _stepStartTime;
    private int   _stepErrors;

    void Start()
    {
        SessionManager.Instance.TransitionToPhase("tutorial");
        BeginStep(TutStep.MoveTo);
    }

    public void OnMovementComplete()  => CompleteStep(TutStep.GrabShovel);
    public void OnShovelGrabbed()     => CompleteStep(TutStep.OpenDoor);
    public void OnDoorOpened()        => CompleteStep(TutStep.Complete);

    private void BeginStep(TutStep step)
    {
        currentStep = step;
        _stepStartTime = Time.time;
        _stepErrors = 0;
        instructionUI.ShowStep(step.ToString());
        TelemetryManager.Instance.LogEvent("tutorial_step_begin",
            new { step = step.ToString() });
    }

    private void CompleteStep(TutStep next)
    {
        TelemetryManager.Instance.LogEvent("tutorial_step_complete", new {
            step        = currentStep.ToString(),
            duration    = Time.time - _stepStartTime,
            errorCount  = _stepErrors
        });
        if (next == TutStep.Complete) OnTutorialComplete();
        else BeginStep(next);
    }

    private void OnTutorialComplete()
    {
        TelemetryManager.Instance.LogEvent("task_complete", new {
            task     = "tutorial",
            duration = SessionManager.Instance.GetPhaseDuration()
        });
        SceneTransitionManager.Instance.LoadRoom2();
    }
}

Room 2 Low Cognitive / HieroglyphPuzzleManager.cs

HieroglyphPuzzleManager

Player taps hieroglyph tiles to spell Tutankhamun. Validates sequence, logs every selection.
public class HieroglyphPuzzleManager : MonoBehaviour
{
    // ── Config ──────────────────────────────────────
    public HieroglyphTile[]  allTiles;
    public string[]           correctSequence; // e.g. {"tut","ankh","amun"}
    public PlankBridgeManager bridgeManager;
    public TaskInstructionUI  ui;

    // ── State ────────────────────────────────────────
    private List<string> _selectedSequence = new();
    private int          _wrongAttempts;
    private float        _taskStart;

    void Start()
    {
        _taskStart = Time.time;
        TelemetryManager.Instance.LogEvent("puzzle_start",
            new { puzzle = "hieroglyph" });
        ui.Show("Translate the cartouche to reveal the king's name");
    }

    // Called by HieroglyphTile on select
    public void OnTileSelected(string glyphId)
    {
        _selectedSequence.Add(glyphId);
        TelemetryManager.Instance.LogEvent("hieroglyph_selected", new {
            glyphId,
            positionInSequence = _selectedSequence.Count - 1,
            timestamp = Time.time
        });

        if (_selectedSequence.Count == correctSequence.Length)
            ValidateSequence();
    }

    private void ValidateSequence()
    {
        bool correct = Enumerable.SequenceEqual(_selectedSequence, correctSequence);
        TelemetryManager.Instance.LogEvent("puzzle_attempt", new {
            puzzle       = "hieroglyph",
            correct,
            sequence     = string.Join(",", _selectedSequence),
            attemptCount = ++_wrongAttempts + (correct ? -1 : 0),
            timeElapsed  = Time.time - _taskStart
        });

        if (correct) OnPuzzleSolved();
        else { _selectedSequence.Clear(); _wrongAttempts++; ResetTiles(); }
    }

    private void OnPuzzleSolved()
    {
        TelemetryManager.Instance.LogEvent("task_complete", new {
            task       = "hieroglyph_puzzle",
            duration   = Time.time - _taskStart,
            errorCount = _wrongAttempts
        });
        bridgeManager.EnablePlanks(); // unlock bridge task
    }

    private void ResetTiles() { foreach (var t in allTiles) t.Deselect(); }
}

Room 2 Low Cognitive / PlankBridgeManager.cs

PlankBridgeManager

Player places 3 planks across a pit. Unlocked after puzzle. Tracks placement order and errors.
public class PlankBridgeManager : MonoBehaviour
{
    public PlankInteractable[] planks;     // 3 planks in scene
    public Transform[]         snapZones;  // 3 target positions on bridge
    public GameObject          bridgeBlocker; // invisible wall until complete
    public TaskInstructionUI   ui;

    private int   _planksPlaced;
    private float _taskStart;

    public void EnablePlanks()
    {
        _taskStart = Time.time;
        foreach (var p in planks) p.SetInteractable(true);
        ui.Show("Place the wooden planks to cross the chasm");
        TelemetryManager.Instance.LogEvent("task_start",
            new { task = "bridge_building" });
    }

    public void OnPlankPlaced(int plankIndex, float distError, float angError)
    {
        _planksPlaced++;
        TelemetryManager.Instance.LogEvent("plank_placed", new {
            plankIndex, _planksPlaced,
            distanceError = distError, angularError = angError,
            timeElapsed   = Time.time - _taskStart
        });

        if (_planksPlaced == planks.Length) OnBridgeComplete();
    }

    private void OnBridgeComplete()
    {
        bridgeBlocker.SetActive(false);
        TelemetryManager.Instance.LogEvent("task_complete", new {
            task = "bridge_building", duration = Time.time - _taskStart
        });
        SceneTransitionManager.Instance.LoadRoom3();
    }
}

PlankInteractable

Plank prefab. Extends SnapInteractor logic with bridge-specific reporting.
public class PlankInteractable : MonoBehaviour
{
    public int                 plankIndex;    // 0, 1, 2
    public SnapInteractor     snapInteractor;
    public PlankBridgeManager bridgeManager;
    public Transform          snapZone;

    private bool _placed;

    public void SetInteractable(bool val) =>
        GetComponent<XRGrabInteractable>().enabled = val;

    // Fires from XRGrabInteractable → SelectExited
    public void OnRelease()
    {
        if (_placed) return;
        snapInteractor.OnReleased(); // handles snap + telemetry

        if (distErr < 0.4f) // generous zone for planks
        {
            _placed = true;
            bridgeManager.OnPlankPlaced(plankIndex, distErr,
                Quaternion.Angle(transform.rotation, snapZone.rotation));
        }
    }
}

TreasureAssemblyManager

5-step ritual: place treasures in correct sequence on altar. Any wrong order resets the step. High overload expected in Baseline.
public class TreasureAssemblyManager : MonoBehaviour
{
    // ── Sequence config ─────────────────────────────
    public string[]          correctOrder = {
        "canopic_jar", "ankh_sceptre", "ushabti", "scarab", "crown"
    };

    public TreasureObject[] treasures;        // all 5 objects
    public AssemblySnapZone[] snapZones;       // 5 altar positions
    public TaskInstructionUI ui;

    // ── Runtime state ────────────────────────────────
    private int   _currentStep;
    private int   _sequenceErrors;
    private float _taskStart;
    private float _stepStart;

    void Start()
    {
        _taskStart = _stepStart = Time.time;
        SessionManager.Instance.TransitionToPhase("procedural");
        TelemetryManager.Instance.LogEvent("task_start",
            new { task = "treasure_assembly", requiredSteps = correctOrder.Length });
        ShowCurrentStepInstruction();
    }

    // Called by AssemblySnapZone when an object lands
    public void OnObjectPlaced(string objectId, float distErr, float angErr)
    {
        bool correctObj = objectId == correctOrder[_currentStep];

        TelemetryManager.Instance.LogEvent("treasure_placed", new {
            objectId, step = _currentStep, correct = correctObj,
            distanceError = distErr, angularError = angErr,
            stepDuration  = Time.time - _stepStart,
            cumulativeErrors = _sequenceErrors
        });

        if (correctObj)
        {
            _currentStep++;
            _stepStart = Time.time;
            if (_currentStep >= correctOrder.Length) OnAssemblyComplete();
            else ShowCurrentStepInstruction();
        }
        else
        {
            _sequenceErrors++;
            TelemetryManager.Instance.LogEvent("sequence_error", new {
                expected = correctOrder[_currentStep], received = objectId,
                errorCount = _sequenceErrors
            });
            FeedbackManager.Instance.Trigger(
                SessionManager.Instance.activeProfile.useHaptics,
                null, null, 0.3f, false); // error pulse
        }
    }

    private void ShowCurrentStepInstruction()
    {
        string[] labels = { "Canopic Jar","Ankh Sceptre","Ushabti","Scarab","Crown" };
        ui.Show($"Step {_currentStep+1}/5 — Place the {labels[_currentStep]}");
    }

    private void OnAssemblyComplete()
    {
        TelemetryManager.Instance.LogEvent("task_complete", new {
            task           = "treasure_assembly",
            totalDuration  = Time.time - _taskStart,
            sequenceErrors = _sequenceErrors,
            profile        = SessionManager.Instance.activeProfile.profileName
        });
        SceneTransitionManager.Instance.EndSession();
    }
}

UI / FixedHUDController.cs

FixedHUDController

Baseline only. World-space canvas fixed in front of player. Tracks how often player looks away — key split-attention metric.
public class FixedHUDController : MonoBehaviour
{
    public Transform playerCamera;
    public Vector3   worldPosition;   // fixed in scene, NOT head-locked
    public TMP_Text  taskLabel;

    // Gaze tracking for split-attention analysis
    private bool  _isLookingAtHUD;
    private int   _gazeShiftCount;
    private float _lastGazeTime;
    private float _gazeAngleThreshold = 15f;

    void Update() => TrackGaze();

    private void TrackGaze()
    {
        Vector3 toHUD = (transform.position - playerCamera.position).normalized;
        float angle = Vector3.Angle(playerCamera.forward, toHUD);
        bool lookingNow = angle < _gazeAngleThreshold;

        if (lookingNow != _isLookingAtHUD)
        {
            _isLookingAtHUD = lookingNow;
            if (lookingNow)
            {
                _gazeShiftCount++;
                TelemetryManager.Instance.LogEvent("hud_gaze_on", new {
                    gazeShiftNumber = _gazeShiftCount,
                    timeSinceLastGaze = Time.time - _lastGazeTime
                });
                _lastGazeTime = Time.time;
            }
            else
            {
                TelemetryManager.Instance.LogEvent("hud_gaze_off", new {
                    gazeDuration = Time.time - _lastGazeTime
                });
            }
        }
    }
}
